export default async function handler(req, res) { res.setHeader('Access-Control-Allow-Origin', 'https://www.habitoangular.com'); res.setHeader('Access-Control-Allow-Methods', 'POST, OPTIONS'); res.setHeader('Access-Control-Allow-Headers', 'Content-Type'); if (req.method === 'OPTIONS') return res.status(200).end(); if (req.method !== 'POST') return res.status(405).json({ error: 'Método não permitido' }); try { const { habito, tentativas, porQueAgora, investimento, timestamp } = req.body; const pontuacao = calcularPontuacao({ habito, tentativas, porQueAgora, investimento }); const payload = { habito, tentativas, porQueAgora, investimento, timestamp: timestamp || new Date().toISOString(), pontos: pontuacao.pontos, classificacao: pontuacao.classificacao }; const makeResponse = await fetch('https://hook.us2.make.com/gb72g5x5ek99b79hiqi1ecv6vjjl6qxn', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(payload) }); if (!makeResponse.ok) throw new Error('Erro ao enviar para o Make'); return res.status(200).json({ success: true }); } catch (error) { console.error('Erro:', error); return res.status(500).json({ error: 'Erro interno' }); } } function calcularPontuacao({ habito, tentativas, porQueAgora, investimento }) { let pontos = 0; if (habito === 'exercicio' || habito === 'devocional') pontos++; if (tentativas === '3+') pontos++; if (porQueAgora && porQueAgora.length > 80) pontos++; if (investimento === 'sim') pontos++; const classificacao = pontos >= 3 ? '🔥 QUENTE' : pontos === 2 ? '🌡 MORNO' : '❄️ FRIO'; return { pontos, classificacao }; }
